```
function simpleHash(str: string): string {
  let hash = 5381;                    // Magic number
  
  for (char in str) {
    hash = ((hash << 5) + hash) + char;  // hash * 33 + char
    hash = hash & hash;               // Convert to 32-bit
  }
  
  return hash.toString(16);           // Hex output
}
```
